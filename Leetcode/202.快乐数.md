```
class Solution:
    def isHappy(self, n):
        """
        :type n: int
        :rtype: bool
        """
        def squareSum(num):
            res = 0
            for i in str(num):
                res += int(i) ** 2
            return res
                
        lookup = set()
        happy_sum = squareSum(n)
        while happy_sum not in lookup and happy_sum != 1:
            lookup.add(happy_sum)
            happy_sum = squareSum(happy_sum)
        return True if happy_sum == 1 else False

```
