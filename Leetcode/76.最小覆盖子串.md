## 最小覆盖子串
### 解题思路
#### 思路一

```
class Solution(object):
    def minWindow(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: str
        """
        # corner cases
        if len(s) < len(t):
            return ''
        if s == '' or t == '':
            return ''
        lookup = collections.Counter(t)
        cnt = len(lookup.keys())
        start, end, head, length = 0, 0, 0, float('inf')
        while end < len(s):
            if s[end] in lookup:
                lookup[s[end]] -= 1
                if lookup[s[end]] == 0:
                    cnt -= 1
            end += 1
            while cnt == 0:
                if s[start] in lookup:
                    lookup[s[start]] += 1
                    if lookup[s[start]] >0:
                        cnt += 1
                    if end - start <length:
                        length = end -start
                        head = start
                start += 1
        if length == float('inf'):
            return ''
        return s[head:head+length]

```
